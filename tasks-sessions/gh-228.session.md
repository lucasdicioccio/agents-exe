```haskell
-- | Register a single SQLite tool with the LLM system.
--
-- SQLite tools expose a single 'query' function that accepts SQL.
-- The function name includes the toolbox name for uniqueness.
registerSqliteTool ::
    SqliteTools.Toolbox ->
    Either String ToolRegistration
registerSqliteTool box =
    let llmName = sqlite2LLMName box (SqliteTools.toolDescription box)
        
        -- Single parameter: the SQL query
        paramProps = 
            [ ParamProperty
                { propertyKey = "sql"
                , propertyType = StringParamType
                , propertyDescription = "SQL query to execute (SELECT for read-only, any valid SQL for read-write)"
                , propertyRequired = True
                }
            ]
        
        openaiTool = OpenAI.Tool
            { OpenAI.toolName = llmName
            , OpenAI.toolDescription = SqliteTools.toolDescription box
            , OpenAI.toolParamProperties = paramProps
            }

        -- Create the tool
        tool' = sqliteTool box

        -- Find function - matches on the LLM name
        find :: OpenAI.ToolCall -> Maybe (Tool OpenAI.ToolCall)
        find call =
            if call.toolCallFunction.toolCallFunctionName == llmName
                then Just $ mapToolResult (const call) tool'
                else Nothing
     in Right $ ToolRegistration
            { innerTool = mapToolResult (const ()) tool'
            , declareTool = openaiTool
            , findTool = find
            }

-- | Register all tools from a SQLite toolbox.
--
-- Currently SQLite toolboxes expose a single query tool.
registerSqliteTools ::
    SqliteTools.Toolbox ->
    IO (Either String [ToolRegistration])
registerSqliteTools box =
    pure $ case registerSqliteTool box of
        Left err -> Left err
        Right reg -> Right [reg]
```

### 4. Add internal tool builder

```haskell
-- | Builder for a tool based on a SQLite toolbox.
sqliteTool ::
    SqliteTools.Toolbox ->
    Tool ()
sqliteTool box =
    Tool
        { toolDef = SqliteTool (SqliteTools.toolDescription box)
        , toolRun = run
        }
  where
    call = ()
    run tracer _ctx (Aeson.Object v) = do
        case Aeson.lookup "sql" v of
            Just (Aeson.String query) -> do
                result <- SqliteTools.executeQuery box query
                case result of
                    Left err -> pure $ SqliteToolError call err
                    Right rsp -> pure $ SqliteToolResult call rsp
            _ -> pure $ SqliteToolError call (SqliteTools.InvalidQueryError "Missing 'sql' parameter")
    run _tracer _ctx _ = do
        pure $ SqliteToolError call (SqliteTools.InvalidQueryError "Arguments must be a JSON object")
```

```haskell
    OpenAI.ToolName (mconcat ["sqlite_", box.toolboxName, "_", toolName])
```
Generates LLM-safe tool names in the format `sqlite_{toolboxName}_{toolName}`.

### 3. Added Registration Functions

**`registerSqliteTool`** - Registers a single SQLite tool with the LLM system:
- Creates a "query" tool that accepts SQL
- Sets up a required `sql` string parameter
- Creates the OpenAI tool declaration
- Returns a `ToolRegistration` with the tool, declaration, and find function

**`registerSqliteTools`** - Registers all tools from a SQLite toolbox:
- Currently returns a single tool since SQLite toolboxes expose only one query tool

### 4. Added Internal Tool Builder

**`sqliteTool`** - Builder for a SQLite-based tool:
- Creates a `Tool` with `SqliteTool` definition
- Extracts the "sql" parameter from JSON input
- Executes the query using `SqliteTools.executeQuery`
- Returns `SqliteToolResult` on success or `SqliteToolError` on failure

Added to the export list:

The implementation follows the established patterns from other tool types (Bash, MCP, OpenAPI, PostgREST) in the module.
