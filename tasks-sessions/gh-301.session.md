## Summary

Implement a `new tool` command that creates new tool scripts with proper boilerplate for various programming languages. This accelerates tool development by generating the `describe|run` protocol skeleton.

## Requirements

```bash
# Create a new bash tool
agents-exe new tool my-search bash ./tools/my-search

# Create a new python tool
agents-exe new tool data-processor python ./tools/data-processor

# Create a new haskell tool
agents-exe new tool validator haskell ./tools/validator
```

data ScriptInfo = ScriptInfo
    { scriptArgs :: [ScriptArg]
    , scriptSlug :: Text
    , scriptDescription :: Text
    , scriptEmptyResultBehavior :: Maybe ScriptEmptyResultBehavior
    } deriving (Show, Eq, Ord)

data ScriptArg = ScriptArg
    { argName :: Text
    , argDescription :: Text
    , argTypeString :: Text
    , argBackingTypeString :: Text
    , argTypeArity :: ScriptArgArity
    , argCallingMode :: ScriptArgCallingMode
    } deriving (Show, Eq, Ord)

data ScriptArgArity = Single | Optional
    deriving (Show, Eq, Ord)

data ScriptArgCallingMode 
    = Stdin 
    | Positional 
    | DashDashSpace 
    | DashDashEqual
    deriving (Show, Eq, Ord)

data ScriptEmptyResultBehavior
    = DoNothing
    | AddMessage Text
    deriving (Show, Eq, Ord)

-- | Render a tool template for the given language and slug
renderToolTemplate :: ToolLanguage -> Text -> ToolTemplate
renderToolTemplate Bash slug = ToolTemplate

    , "        break;"
    , "    case 'run':"

-- | Handle the new tool command
handleNewTool :: NewToolOptions -> IO ()
handleNewTool opts = do
    let template = renderToolTemplate opts.newToolLanguage opts.newToolSlug
    let finalPath = opts.newToolPath ++ Text.unpack template.templateExtension
    
    -- Check if file already exists
    exists <- doesFileExist finalPath
    when exists $ do
        putStrLn $ "Error: File already exists: " ++ finalPath
        exitFailure
    
    -- Create directory
    createDirectoryIfMissing True (takeDirectory finalPath)
    
    -- Write tool file
    Text.writeFile finalPath template.templateContent
    
    -- Make executable if needed
    when template.templateMakeExecutable $ do
        let mode = ownerReadMode `unionFileModes` ownerWriteMode `unionFileModes` ownerExecuteMode
        setFileMode finalPath mode
    
    putStrLn $ "Created tool: " ++ finalPath
    putStrLn $ "Language: " ++ show opts.newToolLanguage
    putStrLn $ "Next steps:"
    putStrLn $ "  1. Edit the 'args' array in the describe function"
    putStrLn $ "  2. Implement the run function"
    putStrLn $ "  3. Test with: agents-exe describe-tool " ++ finalPath
