### Module: json

```lua
local json = require("json")
local encoded = json.encode({foo = "bar", count = 42})  -- → '{"foo":"bar","count":42}'
local decoded = json.decode('{"foo":"bar"}')          -- → {foo = "bar"}
```

Implementation in `System.Agents.Tools.LuaToolbox.Modules.Json`:

```haskell
module System.Agents.Tools.LuaToolbox.Modules.Json where

import qualified Data.Aeson as Aeson
import qualified Data.Aeson.Text as AesonText
import Data.Bifunctor (first)
import Data.Text (Text)
import qualified Data.Text as Text
import qualified Data.Text.Lazy as LazyText
import qualified HsLua as Lua

-- | Register the json module in the Lua state
registerJsonModule :: LuaState -> IO ()
registerJsonModule lstate = Lua.runWith lstate $ do
    Lua.newtable  -- Create module table

    -- json.encode(value) -> string
    Lua.pushstring "encode"
    Lua.pushhsfunction luaEncode
    Lua.settable (Lua.nthFromTop 3)

    -- json.decode(string) -> value
    Lua.pushstring "decode"
    Lua.pushhsfunction luaDecode
    Lua.settable (Lua.nthFromTop 3)

    -- Register as 'json' module
    Lua.setglobal "json"

luaEncode :: Lua.LuaE e Lua.NumResults
luaEncode = do
    -- Convert Lua value to Aeson Value, then to string
    jsonVal <- luaToAesonValue
    let txt = LazyText.toStrict $ AesonText.encodeToLazyText jsonVal
    Lua.pushstring (Text.encodeUtf8 txt)
    pure 1

luaDecode :: Lua.LuaE e Lua.NumResults
luaDecode = do
    -- Get string argument
    str <- Lua.tostring' Lua.stackTop
    Lua.pop 1
    case Aeson.eitherDecodeStrict' str of
        Left err -> do
            Lua.pushnil
            Lua.pushstring (Text.encodeUtf8 $ Text.pack err)
            pure 2  -- nil, error
        Right val -> do
            aesonToLuaValue val
            pure 1
```

```haskell
module System.Agents.Tools.LuaToolbox.Modules.Time where

import Data.Time (UTCTime, defaultTimeLocale, formatTime, getCurrentTime)
import Data.Time.Clock.POSIX (utcTimeToPOSIXSeconds, posixSecondsToUTCTime)
import qualified HsLua as Lua

registerTimeModule :: LuaState -> IO ()
registerTimeModule lstate = Lua.runWith lstate $ do
    Lua.newtable

    addFunction "now" luaNow
    addFunction "sleep" luaSleep
    addFunction "format" luaFormat
    addFunction "diff" luaDiff

    Lua.setglobal "time"

luaNow :: Lua.LuaE e Lua.NumResults
luaNow = do
    now <- liftIO getCurrentTime
    let seconds = realToFrac $ utcTimeToPOSIXSeconds now :: Double
    Lua.pushnumber seconds
    pure 1

luaSleep :: Lua.LuaE e Lua.NumResults
luaSleep = do
    seconds <- Lua.tonumber Lua.stackTop
    Lua.pop 1
    liftIO $ threadDelay (round $ seconds * 1000000)
    pure 0
```

### Module: fs (sandboxed filesystem)

```lua
local fs = require("fs")
fs.read("/path/to/file.txt")           -- → content string or nil, error
fs.write("/path/to/file.txt", "data")  -- → true or false, error
fs.exists("/path/to/file.txt")         -- → true/false
fs.list("/path/to/dir")                -- → {names} or nil, error
fs.mkdir("/path/to/dir")               -- → true or false, error
```

Implementation:

```haskell
module System.Agents.Tools.LuaToolbox.Modules.Fs where

import System.Directory (createDirectory, doesDirectoryExist, doesFileExist, listDirectory)
import System.FilePath (isAbsolute, normalise, (</>))
import qualified Data.ByteString as BS
import qualified Data.Text as Text
import qualified Data.Text.Encoding as Text
import qualified HsLua as Lua

-- | Filesystem module with path sandboxing
data FsConfig = FsConfig
    { fsAllowedPaths :: [FilePath]
    -- ^ Whitelist of allowed paths (if empty, all paths allowed - DANGEROUS)
    }

registerFsModule :: LuaState -> FsConfig -> IO ()
registerFsModule lstate config = Lua.runWith lstate $ do
    -- Store config in registry for access from Haskell functions
    Lua.pushlightuserdata (castPtr config)  -- unsafe, needs proper stable pointer
    Lua.setfield Lua.registryindex "__FS_CONFIG"

    Lua.newtable

    addFunction "read" (luaRead config)
    addFunction "write" (luaWrite config)
    addFunction "exists" (luaExists config)
    addFunction "list" (luaList config)
    addFunction "mkdir" (luaMkdir config)

    Lua.setglobal "fs"

-- | Validate and normalize path against allowed paths
validatePath :: FsConfig -> FilePath -> Either Text FilePath
validatePath config path
    | null (fsAllowedPaths config) = Right normalised
    | otherwise = if any (`isPathPrefixOf` normalised) (fsAllowedPaths config)
        then Right normalised
        else Left "Path not in allowed paths"
  where
    normalised = normalise path
    isPathPrefixOf prefix p = prefix `isPrefixOf` p || (prefix ++ "/") `isPrefixOf` p

luaRead :: FsConfig -> Lua.LuaE e Lua.NumResults
luaRead config = do
    path <- Lua.tostring' Lua.stackTop
    Lua.pop 1

    case validatePath config (Text.unpack $ Text.decodeUtf8 path) of
        Left err -> do
            Lua.pushnil
            Lua.pushstring (Text.encodeUtf8 err)
            pure 2
        Right validPath -> do
            result <- liftIO $ try $ BS.readFile validPath
            case result of
                Left (e :: IOException) -> do
                    Lua.pushnil
                    Lua.pushstring (Text.encodeUtf8 $ Text.pack $ show e)
                    pure 2
                Right content -> do
                    Lua.pushstring content
                    pure 1
```

```haskell
module System.Agents.Tools.LuaToolbox.Modules.Http where

import qualified Network.HTTP.Client as HTTP
import qualified Network.HTTP.Client.TLS as HTTPS
import Network.HTTP.Types (Status, statusCode)
import qualified HsLua as Lua

data HttpConfig = HttpConfig
    { httpAllowedHosts :: [Text]
    -- ^ Whitelist of allowed hosts (if empty, all hosts blocked)
    }

registerHttpModule :: LuaState -> HttpConfig -> IO ()
registerHttpModule lstate config = do
    manager <- HTTPS.newTlsManager
    Lua.runWith lstate $ do
        -- Store manager and config in registry
        Lua.pushlightuserdata (castPtr manager)
        Lua.setfield Lua.registryindex "__HTTP_MANAGER"

        Lua.newtable
        addFunction "get" (luaGet config manager)
        addFunction "post" (luaPost config manager)
        addFunction "request" (luaRequest config manager)
        Lua.setglobal "http"

validateHost :: HttpConfig -> Text -> Either Text ()
validateHost config host
    | null (httpAllowedHosts config) = Left "No hosts allowed"
    | host `elem` httpAllowedHosts config = Right ()
    | otherwise = Left "Host not in allowed list"
```

## Tradeoffs Documented

1. **HTTP uses direct client**: Using http-client directly rather than through the toolbox for simplicity. Could be changed to use toolbox's http tools in the future.
2. **Path validation**: Simple prefix matching for path validation. May not catch all edge cases (symlinks, etc.).
3. **No async HTTP**: HTTP operations are blocking. Lua coroutines can be used for cooperative multitasking.
4. **JSON numbers**: All JSON numbers become Lua numbers (doubles). Large integers may lose precision.
