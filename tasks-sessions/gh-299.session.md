Implement a standalone `describe` command that validates and displays the `describe` registration of a tool script without needing to create an agent file. This enables rapid iteration when developing new tools.

```haskell
{-# LANGUAGE OverloadedRecordDot #-}
{-# LANGUAGE OverloadedStrings #-}

module System.Agents.CLI.DescribeTool (
    handleDescribeTool,
    DescribeToolOptions(..),
) where

import qualified Data.Aeson as Aeson
import qualified Data.Aeson.Encode.Pretty as Aeson
import qualified Data.ByteString.Lazy as LByteString
import Data.Text (Text)
import qualified Data.Text as Text
import qualified Data.Text.IO as Text
import System.Exit (exitFailure)

import qualified System.Agents.Tools.Bash as BashTools

-- | Options for the describe-tool command
data DescribeToolOptions = DescribeToolOptions
    { describeToolPath :: FilePath
    -- ^ Path to the tool script to validate
    } deriving (Show)

-- | Handle the describe-tool command
handleDescribeTool :: DescribeToolOptions -> IO ()
handleDescribeTool opts = do
    result <- loadAndValidateTool opts.describeToolPath
    case result of
        Left err -> do
            Text.hPutStrLn stderr $ formatError err
            exitFailure
        Right info -> do
            LByteString.putStrLn $ Aeson.encodePretty info

-- | Load and validate a tool script
type LoadResult = Either BashTools.InvalidScriptError BashTools.ScriptInfo

loadAndValidateTool :: FilePath -> IO LoadResult
loadAndValidateTool path = do
    -- Use loadScript from Bash module which invokes 'describe' command
    result <- BashTools.loadScript silentTracer path
    case result of
        Left err -> pure $ Left err
        Right desc -> pure $ Right desc.scriptInfo
  where
    silentTracer = Prod.Tracer (const (pure ()))

-- | Format an error for display
formatError :: BashTools.InvalidScriptError -> Text
formatError (BashTools.InvalidScriptError path code stderr) = Text.unlines
    [ "Error: Tool script failed during describe"
    , "  Path: " <> Text.pack path
    , "  Exit code: " <> Text.pack (show code)
    , "  Stderr: " <> Text.decodeUtf8 stderr
    ]
formatError (BashTools.InvalidDescriptionError path jsonErr) = Text.unlines
    [ "Error: Tool returned invalid JSON description"
    , "  Path: " <> Text.pack path
    , "  Parse error: " <> Text.pack jsonErr
    , ""
    , "Hint: Run the tool with 'describe' argument manually to see its output:"
    , "  " <> Text.pack path <> " describe"
    ]
```
