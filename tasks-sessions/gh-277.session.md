```haskell
module System.Agents.ToolRegistration where

-- ... existing imports ...
import qualified System.Agents.Tools.LuaToolbox as LuaTools

-- naming policy for Lua tools
lua2LLMName :: LuaTools.Toolbox -> Text -> OpenAI.ToolName
lua2LLMName box toolName =
    OpenAI.ToolName (mconcat ["lua_", box.toolboxName, "_", toolName])

-- | Register a Lua toolbox with the LLM system.
--
-- Lua toolboxes expose:
-- * execute: Execute arbitrary Lua code
-- * file: Execute Lua code from a file (if allowed paths permit)
--
-- The 'execute' tool is always available. The 'file' tool is available
-- if allowedPaths is non-empty.
registerLuaTool ::
    LuaTools.Toolbox ->
    Either String ToolRegistration
registerLuaTool box =
    let toolName = "execute"
        llmName = lua2LLMName box toolName

        paramProps =
            [ ParamProperty
                { propertyKey = "script"
                , propertyType = StringParamType
                , propertyDescription = "Lua script source code to execute"
                , propertyRequired = True
                }
            , ParamProperty
                { propertyKey = "timeout"
                , propertyType = NumberParamType
                , propertyDescription = "Optional timeout override in seconds (default: toolbox configured timeout)"
                , propertyRequired = False
                }
            ]

        openaiTool =
            OpenAI.Tool
                { OpenAI.toolName = llmName
                , OpenAI.toolDescription = box.toolboxConfig.luaToolboxDescription
                , OpenAI.toolParamProperties = paramProps
                }

        -- Create the tool
        tool' = luaTool box

        find :: OpenAI.ToolCall -> Maybe (Tool OpenAI.ToolCall)
        find call =
            if call.toolCallFunction.toolCallFunctionName == llmName
                then Just $ mapToolResult (const call) tool'
                else Nothing
     in
        Right $ ToolRegistration
            { innerTool = mapToolResult (const ()) tool'
            , declareTool = openaiTool
            , findTool = find
            }

-- | Builder for Lua toolbox tool
luaTool :: LuaTools.Toolbox -> Tool ()
luaTool box =
    Tool
        { toolDef = LuaTool box.toolboxName
        , toolRun = run
        }
  where
    call = ()
    run tracer ctx (Aeson.Object v) = do
        -- Extract script from arguments
        case KeyMap.lookup (AesonKey.fromText "script") v of
            Just (Aeson.String script) -> do
                -- Get portal and allowed tools from context
                let mPortal = ctxToolPortal ctx
                let allowedTools = ctxAllowedTools ctx

                -- Execute script with portal
                result <- LuaTools.executeScriptWithPortal
                    box
                    script
                    (contramap LuaToolsTrace tracer)
                    mPortal
                    allowedTools

                case result of
                    Left err -> pure $ LuaToolError call (Text.pack $ show err)
                    Right execResult -> pure $ LuaToolResult call (Aeson.toJSON execResult)

            _ ->
                pure $ LuaToolError call "Missing 'script' parameter or invalid type"

    run _tracer _ctx _ =
        pure $ LuaToolError call "Arguments must be a JSON object"
```

### 2. Create the Portal Implementation

Create `System.Agents.Tools.Portal` that implements the actual tool dispatch:

```haskell
{-# LANGUAGE OverloadedRecordDot #-}

module System.Agents.Tools.Portal where

import Control.Exception (try)
import qualified Data.Aeson as Aeson
import Data.Map.Strict (Map)
import qualified Data.Map.Strict as Map
import Data.Text (Text)
import qualified Data.Text as Text
import Data.Time (diffUTCTime, getCurrentTime)
import Data.UUID.V4 (nextRandom)
import Data.UUID (toText)

import System.Agents.Tools.Base (CallResult(..), Tool(..))
import System.Agents.Tools.Context (ToolCall(..), ToolPortal, ToolResult(..), TraceId)
import qualified System.Agents.LLMs.OpenAI as OpenAI

-- | A registered tool that can be called through the portal
data RegisteredTool = RegisteredTool
    { regToolName :: Text
    , regToolHandler :: Aeson.Value -> IO (Either Text Aeson.Value)
    }

-- | Build a portal from a list of registered tools
--
-- This creates the callback function that Lua (or other tools) use
-- to invoke other tools.
buildPortal ::
    [RegisteredTool] ->  -- ^ Available tools
    ToolPortal
buildPortal tools call = do
    case findTool (callToolName call) tools of
        Nothing -> do
            pure $ ToolResult
                { resultData = Aeson.object ["error" .= ("Tool not found: " <> callToolName call)]
                , resultDuration = 0
                , resultTraceId = ""
                }
        Just tool -> do
            traceId <- toText <$> nextRandom
            startTime <- getCurrentTime

            result <- try $ regToolHandler tool (callArgs call)

            endTime <- getCurrentTime
            let duration = diffUTCTime endTime startTime

            case result of
                Left (e :: SomeException) ->
                    pure $ ToolResult
                        { resultData = Aeson.object ["error" .= Text.pack (show e)]
                        , resultDuration = duration
                        , resultTraceId = traceId
                        }
                Right (Left err) ->
                    pure $ ToolResult
                        { resultData = Aeson.object ["error" .= err]
                        , resultDuration = duration
                        , resultTraceId = traceId
                        }
                Right (Right val) ->
                    pure $ ToolResult
                        { resultData = val
                        , resultDuration = duration
                        , resultTraceId = traceId
                        }

findTool :: Text -> [RegisteredTool] -> Maybe RegisteredTool
findTool name = find ((== name) . regToolName)

-- | Convert a ToolRegistration to a RegisteredTool for the portal
registrationToTool :: ToolRegistration -> RegisteredTool
registrationToTool reg = RegisteredTool
    { regToolName = getToolName $ declareTool reg
    , regToolHandler = \args -> do
        -- Create a mock tool call to execute the tool
        let mockCall = OpenAI.ToolCall
                { toolCallId = OpenAI.ToolCallId "portal"
                , toolCallFunction = OpenAI.ToolCallFunction
                    { toolCallFunctionName = toolName $ declareTool reg
                    , toolCallFunctionArguments = args
                    }
                }

        case findTool mockCall reg of
            Nothing -> pure $ Left "Tool not found in registration"
            Just tool -> do
                -- Execute the tool (simplified - needs proper context)
                result <- toolRun tool (Tracer $ const $ pure ()) minimalContext args
                case result of
                    BlobToolSuccess _ bs ->
                        pure $ Right $ Aeson.object ["output" .= Text.decodeUtf8 bs]
                    McpToolResult _ mcpResult ->
                        pure $ Right $ Aeson.toJSON mcpResult
                    SqliteToolResult _ queryResult ->
                        pure $ Right $ Aeson.toJSON queryResult
                    SystemToolResult _ sysResult ->
                        pure $ Right $ Aeson.toJSON sysResult
                    -- Handle other result types...
                    err ->
                        pure $ Left $ Text.pack $ show err
    }
  where
    minimalContext = mkMinimalContext
        (SessionId $ error "mock session")
        (ConversationId $ error "mock conversation")
        (TurnId $ error "mock turn")

getToolName :: OpenAI.Tool -> Text
getToolName tool = case toolName tool of
    OpenAI.ToolName t -> t
```

## Tradeoffs Documented

1. **Synchronous portal implementation**: Tool calls block Lua execution. This is simpler and matches the design spec.
2. **Portal recreated per execution**: The portal is built fresh each time. Could be cached for performance, but recreation ensures fresh state.
3. **Tool name matching**: Uses string names for lookup. Could use more type-safe approach, but string names match how tools are declared.
