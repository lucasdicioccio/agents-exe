```haskell
{-# LANGUAGE OverloadedStrings #-}

module System.Agents.Tools.LuaToolbox.Modules.Fs where

import System.Directory (createDirectory, doesDirectoryExist, doesFileExist, listDirectory, canonicalizePath)
import System.FilePath (isAbsolute, normalise, takeDirectory, (</>))
import System.IO.Error (isPermissionError)
import qualified Data.ByteString as BS
import Data.Text (Text)
import qualified Data.Text as Text
import qualified Data.Text.Encoding as Text
import qualified HsLua as Lua

-- | Filesystem configuration with sandboxing
data FsConfig = FsConfig
    { fsAllowedPaths :: [FilePath]
    -- ^ Whitelist of allowed paths. If empty, NO paths are allowed (secure default).
    -- Paths should be absolute and canonical.
    }

-- | Path validation error
data PathError
    = PathNotAllowed FilePath
    | PathNotAbsolute FilePath
    | PathOutsideSandbox FilePath FilePath
    | PathIOError FilePath String
    deriving (Show, Eq)

-- | Validate and canonicalize a path against allowed paths
--
-- Returns the canonical path if valid, or an error.
validatePath :: FsConfig -> FilePath -> IO (Either PathError FilePath)
validatePath config path = do
    -- Path must be absolute
    if not (isAbsolute path)
        then pure $ Left $ PathNotAbsolute path
        else do
            -- Canonicalize to resolve symlinks, .., etc.
            canonical <- canonicalizePath path

            -- Check against allowed paths
            if null (fsAllowedPaths config)
                then pure $ Left $ PathNotAllowed canonical
                else if any (isPathWithin canonical) (fsAllowedPaths config)
                    then pure $ Right canonical
                    else pure $ Left $ PathOutsideSandbox canonical (show $ fsAllowedPaths config)
  where
    -- Check if child is within parent (handles path separator edge cases)
    isPathWithin child parent =
        let parent' = if last parent == '/' then parent else parent ++ "/"
        in child == parent || take (length parent') child == parent'

-- | Enhanced fs.read with proper error handling
luaRead :: FsConfig -> Lua.LuaE e Lua.NumResults
luaRead config = do
    pathBytes <- Lua.tostring' Lua.stackTop
    Lua.pop 1

    let path = Text.unpack $ Text.decodeUtf8 pathBytes

    validation <- liftIO $ validatePath config path
    case validation of
        Left err -> do
            Lua.pushnil
            Lua.pushstring (Text.encodeUtf8 $ Text.pack $ show err)
            pure 2
        Right validPath -> do
            result <- liftIO $ try $ BS.readFile validPath
            case result of
                Left (e :: IOException) -> do
                    Lua.pushnil
                    Lua.pushstring (Text.encodeUtf8 $ Text.pack $ show e)
                    pure 2
                Right content -> do
                    Lua.pushstring content
                    pure 1

-- | fs.patch(path, search, replace) for code patching
luaPatch :: FsConfig -> Lua.LuaE e Lua.NumResults
luaPatch config = do
    pathBytes <- Lua.tostring' (Lua.nthFromTop 3)
    searchBytes <- Lua.tostring' (Lua.nthFromTop 2)
    replaceBytes <- Lua.tostring' (Lua.nthFromTop 1)
    Lua.pop 3

    let path = Text.unpack $ Text.decodeUtf8 pathBytes
    let search = Text.decodeUtf8 searchBytes
    let replace = Text.decodeUtf8 replaceBytes

    validation <- liftIO $ validatePath config path
    case validation of
        Left err -> do
            Lua.pushboolean False
            Lua.pushstring (Text.encodeUtf8 $ Text.pack $ show err)
            Lua.pushnil  -- no diff
            pure 3
        Right validPath -> do
            result <- liftIO $ try $ do
                content <- Text.readFile validPath
                let (newContent, count) = Text.replace search replace content
                if count > 0
                    then do
                        Text.writeFile validPath newContent
                        let diff = generateDiff search replace content
                        pure (True, count, diff)
                    else
                        pure (False, 0, Text.empty)
            case result of
                Left (e :: IOException) -> do
                    Lua.pushboolean False
                    Lua.pushstring (Text.encodeUtf8 $ Text.pack $ show e)
                    Lua.pushnil
                    pure 3
                Right (success, count, diff) -> do
                    Lua.pushboolean success
                    if success
                        then do
                            Lua.pushstring ""
                            Lua.pushstring (Text.encodeUtf8 $ Text.pack $ show diff)
                        else do
                            Lua.pushstring "Pattern not found"
                            Lua.pushnil
                    pure 3

-- | Generate a simple unified diff for the patch
generateDiff :: Text -> Text -> Text -> Text
generateDiff old new content =
    Text.unlines
        [ "--- a/original"
        , "+++ b/modified"
        , "@@ -1 +1 @@"
        , "-" <> old
        , "+" <> new
        ]
```

```haskell
module System.Agents.Tools.LuaToolbox.Modules.Http where

import Network.HTTP.Client (HttpException(..), HttpExceptionContent(..), Request(..), Response(..), method, parseRequest, httpLbs, responseBody, responseStatus, responseHeaders)
import Network.HTTP.Client.TLS (newTlsManager)
import Network.HTTP.Types (Status, hContentType, statusCode)
import qualified Data.Aeson as Aeson
import qualified Data.ByteString.Lazy as LBS
import Data.Text (Text)
import qualified Data.Text as Text
import qualified Data.Text.Encoding as Text
import Network.URI (URI(..), parseURI, uriAuthority, uriRegName)
import qualified HsLua as Lua

data HttpConfig = HttpConfig
    { httpAllowedHosts :: [Text]
    -- ^ Whitelist of allowed hosts. If empty, NO hosts are allowed (secure default).
    }

-- | Extract host from URL
extractHost :: Text -> Either Text Text
extractHost url =
    case parseURI (Text.unpack url) of
        Nothing -> Left "Invalid URL"
        Just uri -> case uriAuthority uri of
            Nothing -> Left "No host in URL"
            Just auth -> Right $ Text.pack $ uriRegName auth

-- | Validate host against whitelist
validateHost :: HttpConfig -> Text -> Either Text ()
validateHost config host
    | null (httpAllowedHosts config) = Left "No hosts allowed (empty whitelist)"
    | host `elem` httpAllowedHosts config = Right ()
    | otherwise = Left $ "Host not in allowed list: " <> host

-- | Enhanced http.get with host validation
luaGet :: HttpConfig -> Manager -> Lua.LuaE e Lua.NumResults
luaGet config manager = do
    urlBytes <- Lua.tostring' (Lua.nthFromTop 2)
    opts <- if Lua.istable (Lua.nthFromTop 1)
        then luaTableToMap
        else pure mempty

    let url = Text.decodeUtf8 urlBytes

    -- Validate host
    case extractHost url of
        Left err -> do
            Lua.pushnil
            Lua.pushstring (Text.encodeUtf8 err)
            pure 2
        Right host ->
            case validateHost config host of
                Left err -> do
                    Lua.pushnil
                    Lua.pushstring (Text.encodeUtf8 err)
                    pure 2
                Right () -> do
                    -- Proceed with request
                    result <- liftIO $ try $ do
                        initReq <- parseRequest (Text.unpack url)
                        let req = initReq
                                { method = "GET"
                                , requestHeaders = buildHeaders opts
                                }
                        response <- httpLbs req manager
                        pure $ responseToTable response
                    case result of
                        Left (e :: HttpException) -> do
                            Lua.pushnil
                            Lua.pushstring (Text.encodeUtf8 $ Text.pack $ show e)
                            pure 2
                        Right tbl -> do
                            pushResponseTable tbl
                            pure 1

buildHeaders :: Map Text Text -> RequestHeaders
buildHeaders opts =
    case HashMap.lookup "headers" opts of
        Just (Aeson.Object hm) ->
            map (\(k, v) -> (CI.mk $ Text.encodeUtf8 k, Text.encodeUtf8 $ AesonText.encodeToLazyText v)) $
            HashMap.toList hm
        _ -> []

responseToTable :: Response LBS.ByteString -> Map Text Aeson.Value
responseToTable resp = HashMap.fromList
    [ ("status", Aeson.Number $ fromIntegral $ statusCode $ responseStatus resp)
    , ("headers", Aeson.Object $ HashMap.fromList $ map (\(k, v) -> (Text.decodeUtf8 $ CI.original k, Aeson.String $ Text.decodeUtf8 v)) $ responseHeaders resp)
    , ("body", Aeson.String $ Text.decodeUtf8 $ LBS.toStrict $ responseBody resp)
    ]
```

Add tracing to `System.Agents.Tools.LuaToolbox.Modules.Tools`:

```haskell
module System.Agents.Tools.LuaToolbox.Modules.Tools where

import Data.Time (getCurrentTime, diffUTCTime)
import Data.UUID (toText)
import Data.UUID.V4 (nextRandom)
import Prod.Tracer (Tracer, runTracer)

-- | Extended tools config with tracer
data ToolsConfig = ToolsConfig
    { toolsPortal :: Maybe ToolPortal
    , toolsAllowed :: [Text]
    , toolsCallerId :: Text
    , toolsTracer :: Tracer IO Trace  -- NEW: for tracing tool calls
    }

-- | Trace events for tool invocations
data Trace
    = ToolCallStarted TraceId Text Text Aeson.Value
    -- ^ traceId, callerId, toolName, args
    | ToolCallCompleted TraceId NominalDiffTime ToolResult
    -- ^ traceId, duration, result
    | ToolCallFailed TraceId Text
    -- ^ traceId, error

-- | Enhanced luaCall with tracing
luaCallWithTracing :: ToolsConfig -> Lua.LuaE e Lua.NumResults
luaCallWithTracing config = do
    toolNameBytes <- Lua.tostring' (Lua.nthFromTop 2)
    let tName = Text.decodeUtf8 toolNameBytes

    argsValue <- if Lua.istable (Lua.nthFromTop 1)
        then do
            luaToAesonValue
        else pure (Aeson.Object mempty)

    Lua.pop 2

    -- Check whitelist
    if not (isToolAllowed' tName config)
        then do
            Lua.pushnil
            Lua.pushstring (Text.encodeUtf8 $ "Tool not allowed: " <> tName)
            pure 2
        else case toolsPortal config of
            Nothing -> do
                Lua.pushnil
                Lua.pushstring "Tool portal not available"
                pure 2
            Just portal -> do
                -- Generate trace ID
                traceId <- liftIO $ toText <$> nextRandom
                startTime <- liftIO getCurrentTime

                -- Trace start
                liftIO $ runTracer (toolsTracer config) $
                    ToolCallStarted traceId (toolsCallerId config) tName argsValue

                let call = ToolCall
                        { callToolName = tName
                        , callArgs = argsValue
                        , callCallerId = toolsCallerId config
                        }

                result <- liftIO $ try $ portal call

                endTime <- liftIO getCurrentTime
                let duration = diffUTCTime endTime startTime

                case result of
                    Left (e :: SomeException) -> do
                        liftIO $ runTracer (toolsTracer config) $
                            ToolCallFailed traceId (Text.pack $ show e)
                        Lua.pushnil
                        Lua.pushstring (Text.encodeUtf8 $ Text.pack $ show e)
                        pure 2
                    Right toolResult -> do
                        liftIO $ runTracer (toolsTracer config) $
                            ToolCallCompleted traceId duration toolResult
                        pushToolResult toolResult { resultTraceId = traceId }
                        pure 1
```

## Acceptance Criteria

- [ ] Path sandboxing uses canonical paths to prevent symlink traversal attacks
- [ ] Empty allowedPaths means NO filesystem access (secure default)
- [ ] Empty allowedHosts means NO network access (secure default)
- [ ] fs.patch() function works for code patching with diff generation
- [ ] HTTP host validation extracts and validates host from URL
- [ ] Tool invocation generates unique trace IDs
- [ ] Tool calls are traced: start, completion, failure
- [ ] Trace ID flows through to ToolResult for call tree reconstruction
- [ ] All IO errors are caught and returned as Lua errors (nil, error)
- [ ] Security tests verify path traversal is blocked
- [ ] Security tests verify host whitelist is enforced

## Tradeoffs Documented

1. **Canonicalization overhead**: Path validation requires canonicalization which involves syscalls. This is necessary for security.
2. **Strict defaults**: Empty whitelist = no access. Users must explicitly allow paths/hosts. More secure but requires configuration.
3. **Trace overhead**: Generating UUIDs and timestamps has small overhead. Acceptable for debugging value.
4. **No fine-grained permissions**: We use whitelists, not capabilities. Can be enhanced later.
