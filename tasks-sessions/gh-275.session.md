```haskell
{-# LANGUAGE OverloadedStrings #-}
{-# LANGUAGE OverloadedRecordDot #-}

module System.Agents.Tools.LuaToolbox.Modules.Tools where

import Control.Concurrent (forkIO, killThread)
import Control.Concurrent.MVar (newEmptyMVar, putMVar, takeMVar)
import Control.Exception (try)
import qualified Data.Aeson as Aeson
import qualified Data.HashMap.Strict as HashMap
import Data.Text (Text)
import qualified Data.Text as Text
import qualified Data.Text.Encoding as Text
import qualified HsLua as Lua

import System.Agents.Tools.Context (ToolCall(..), ToolPortal, ToolResult(..), isToolAllowed)

-- | Configuration for the tools module
data ToolsConfig = ToolsConfig
    { toolsPortal :: Maybe ToolPortal
    -- ^ The portal callback for invoking tools
    , toolsAllowed :: [Text]
    -- ^ Whitelist of allowed tool names
    , toolsCallerId :: Text
    -- ^ Identifier for this Lua script (for tracing)
    }

-- | Register the tools module in the Lua state
registerToolsModule :: LuaState -> ToolsConfig -> IO ()
registerToolsModule lstate config = Lua.runWith lstate $ do
    Lua.newtable  -- Create tools module table

    -- Store config in registry
    pushRegistryTable "__TOOLS_CONFIG" config

    -- tools.call(toolName, args) -> result
    Lua.pushstring "call"
    Lua.pushhsfunction (luaCall config)
    Lua.settable (Lua.nthFromTop 3)

    -- Create convenience wrappers if specific tools are allowed
    -- e.g., tools.bash, tools.sqlite, etc.
    createToolWrappers config

    Lua.setglobal "tools"

-- | Lua function: tools.call(toolName, args)
--
-- Calls a tool through the portal and returns the result.
--
-- Returns: result table or nil, error
-- Result table has: { data = ..., duration = ..., traceId = ... }
luaCall :: ToolsConfig -> Lua.LuaE e Lua.NumResults
luaCall config = do
    -- Get arguments
    toolName <- Lua.tostring' (Lua.nthFromTop 2)
    Lua.pushnil  -- first key for args table

    -- Convert args table to Aeson Value
    argsValue <- if Lua.istable (Lua.nthFromTop 1)
        then do
            Lua.pop 1
            luaToAesonValue  -- This will pop the table
        else do
            Lua.pop 1
            pure Aeson.Object mempty  -- Empty object if no args

    let tName = Text.decodeUtf8 toolName

    -- Check if tool is allowed
    if not (isToolAllowed' tName config)
        then do
            Lua.pushnil
            Lua.pushstring (Text.encodeUtf8 $ "Tool not allowed: " <> tName)
            pure 2
        else case toolsPortal config of
            Nothing -> do
                Lua.pushnil
                Lua.pushstring "Tool portal not available"
                pure 2
            Just portal -> do
                -- Create tool call
                let call = ToolCall
                        { callToolName = tName
                        , callArgs = argsValue
                        , callCallerId = toolsCallerId config
                        }

                -- Execute through portal (this is blocking IO)
                result <- liftIO $ try $ portal call

                case result of
                    Left (e :: SomeException) -> do
                        Lua.pushnil
                        Lua.pushstring (Text.encodeUtf8 $ Text.pack $ show e)
                        pure 2
                    Right toolResult -> do
                        -- Convert ToolResult to Lua table
                        pushToolResult toolResult
                        pure 1

isToolAllowed' :: Text -> ToolsConfig -> Bool
isToolAllowed' toolName config =
    null (toolsAllowed config) || toolName `elem` toolsAllowed config

pushToolResult :: ToolResult -> Lua ()
pushToolResult tr = do
    Lua.newtable

    -- result.data
    Lua.pushstring "data"
    aesonToLuaValue (resultData tr)
    Lua.settable (Lua.nthFromTop 3)

    -- result.duration
    Lua.pushstring "duration"
    Lua.pushnumber (realToFrac $ resultDuration tr)
    Lua.settable (Lua.nthFromTop 3)

    -- result.traceId
    Lua.pushstring "traceId"
    Lua.pushstring (Text.encodeUtf8 $ resultTraceId tr)
    Lua.settable (Lua.nthFromTop 3)

-- | Create convenience wrappers for allowed tools
--
-- e.g., tools.bash.run(), tools.sqlite.query(), etc.
createToolWrappers :: ToolsConfig -> Lua ()
createToolWrappers config = do
    -- Group tools by prefix for nested tables
    let prefixes = ["bash", "sqlite", "postgrest", "system", "io", "mcp"]
    mapM_ (createPrefixWrapper config) prefixes

createPrefixWrapper :: ToolsConfig -> Text -> Lua ()
createPrefixWrapper config prefix = do
    -- Create nested table: tools.bash, tools.sqlite, etc.
    Lua.pushstring (Text.encodeUtf8 prefix)
    Lua.newtable

    -- Add generic call method: tools.bash.call(args)
    Lua.pushstring "call"
    Lua.pushhsfunction (\() -> do
        -- Get self table to find the prefix
        -- This is simplified; actual implementation needs proper closure
        luaCallWithPrefix prefix config)
    Lua.settable (Lua.nthFromTop 3)

    -- Add specific convenience methods based on known tool patterns
    addConvenienceMethods prefix

    Lua.settable (Lua.nthFromTop 3)

-- | Add tool-specific convenience methods
addConvenienceMethods :: Text -> Lua ()
addConvenienceMethods "bash" = do
    -- tools.bash.run(command, opts)
    addMethod "run" luaBashRun
    -- tools.bash.spawn(command)
    addMethod "spawn" luaBashSpawn

addConvenienceMethods "sqlite" = do
    -- tools.sqlite.query(db, query, params)
    addMethod "query" luaSqliteQuery
    -- tools.sqlite.execute(db, query, params)
    addMethod "execute" luaSqliteExecute

addConvenienceMethods "postgrest" = do
    -- tools.postgrest.get(endpoint, opts)
    addMethod "get" luaPostgrestGet
    -- tools.postgrest.post(endpoint, body, opts)
    addMethod "post" luaPostgrestPost
    addMethod "patch" luaPostgrestPatch
    addMethod "delete" luaPostgrestDelete

addConvenienceMethods "system" = do
    -- tools.system.info(capability)
    addMethod "info" luaSystemInfo

addConvenienceMethods _ = pure ()

addMethod :: Text -> Lua.LuaE e Lua.NumResults -> Lua ()
addMethod name fn = do
    Lua.pushstring (Text.encodeUtf8 name)
    Lua.pushhsfunction fn
    Lua.settable (Lua.nthFromTop 3)

-- | tools.bash.run(command, opts)
-- Returns: { exitCode, stdout, stderr }
luaBashRun :: ToolsConfig -> Lua.LuaE e Lua.NumResults
luaBashRun config = do
    -- Extract command and options from arguments
    cmd <- Lua.tostring' (Lua.nthFromTop 2)
    opts <- if Lua.istable (Lua.nthFromTop 1)
        then luaTableToMap
        else pure mempty

    Lua.pop 2

    -- Build tool call
    let args = Aeson.object
            [ "command" .= Text.decodeUtf8 cmd
            , "cwd" .= HashMap.lookup "cwd" opts
            , "env" .= HashMap.lookup "env" opts
            ]

    callTool "bash" args config

-- | Generic tool call helper
callTool :: Text -> Aeson.Value -> ToolsConfig -> Lua.LuaE e Lua.NumResults
callTool toolName args config = do
    if not (isToolAllowed' toolName config)
        then do
            Lua.pushnil
            Lua.pushstring (Text.encodeUtf8 $ "Tool not allowed: " <> toolName)
            pure 2
        else case toolsPortal config of
            Nothing -> do
                Lua.pushnil
                Lua.pushstring "Tool portal not available"
                pure 2
            Just portal -> do
                let call = ToolCall
                        { callToolName = toolName
                        , callArgs = args
                        , callCallerId = toolsCallerId config
                        }
                result <- liftIO $ portal call
                pushToolResult result
                pure 1
```

## Tradeoffs Documented

1. **Synchronous only**: Tool calls are synchronous (blocking). This matches the design spec and simplifies implementation.
2. **No automatic retries**: Failed tool calls return errors; Lua script must handle retry logic.
3. **Table-to-JSON conversion**: Uses simple conversion; complex types may not round-trip perfectly.
4. **Convenience wrappers are static**: Known tool patterns (bash.run, sqlite.query) have wrappers. Unknown tools use generic `call`.
