data Toolbox = Toolbox

{- | Result of a developer operation.
-}
data OperationResult = OperationResult
    { resultCapability :: Text
    , resultData :: Aeson.Value
    , resultExecutionTime :: NominalDiffTime
    } deriving (Show)

instance ToJSON OperationResult where
    toJSON result = Aeson.object
        [ "capability" .= resultCapability result
        , "data" .= resultData result
        , "executionTime" .= formatExecutionTime (resultExecutionTime result)
        ]

{- | Operation errors.
-}
data OperationError
    = CapabilityNotEnabledError !Text
    | InvalidArgumentsError !Text
    | ValidationError !Text
    | ExecutionError !Text
    deriving (Show, Eq)

-- Initialization

initializeToolbox ::
    Tracer IO Trace ->
    DeveloperToolboxDescription ->
    IO (Either String Toolbox)
initializeToolbox _tracer desc = do
    if null desc.developerToolboxCapabilities
        then pure $ Left "Developer toolbox must have at least one capability enabled"
        else pure $ Right $ Toolbox
            { toolboxName = desc.developerToolboxName
            , toolboxDescription = desc.developerToolboxDescription
            , toolboxCapabilities = desc.developerToolboxCapabilities
            }

-- Capability Execution
-------------------------------------------------------------------------------

executeCapability :: Toolbox -> Text -> Aeson.Value -> IO (Either OperationError OperationResult)
executeCapability toolbox capabilityName args = do
    runTracer (Tracer (const (pure ()))) (CapabilityRequestedTrace capabilityName)
    startTime <- getCurrentTime

    case capabilityFromName capabilityName of
        Nothing -> do
            let err = "Unknown capability: " <> capabilityName
            runTracer (Tracer (const (pure ()))) (CapabilityErrorTrace capabilityName err)
            pure $ Left $ ExecutionError err
        Just capability -> do
            if capability `notElem` toolboxCapabilities toolbox
                then pure $ Left $ CapabilityNotEnabledError capabilityName
                else do
                    result <- executeCapabilityInternal capability args
                    endTime <- getCurrentTime
                    let execTime = diffUTCTime endTime startTime
                    case result of
                        Left err -> do
                            runTracer (Tracer (const (pure ()))) (CapabilityErrorTrace capabilityName err
                            pure $ Left $ ExecutionError err
                        Right val -> do
                            runTracer (Tracer (const (pure ()))) (CapabilityCompletedTrace capabilityName execTime)
                            pure $ Right $ OperationResult
                                { resultCapability = capabilityName
                                , resultData = val
                                , resultExecutionTime = execTime
                                }

capabilityFromName :: Text -> Maybe DeveloperCapability
capabilityFromName name = case name of
    "validate-tool" -> Just ValidateToolCapability
    "scaffold-agent" -> Just ScaffoldAgentCapability
    "scaffold-tool" -> Just ScaffoldToolCapability
    "get-spec" -> Just GetSpecCapability
    _ -> Nothing

capabilityToName :: DeveloperCapability -> Text
capabilityToName ValidateToolCapability = "validate-tool"
capabilityToName ScaffoldAgentCapability = "scaffold-agent"
capabilityToName ScaffoldToolCapability = "scaffold-tool"
capabilityToName GetSpecCapability = "get-spec"

executeCapabilityInternal :: DeveloperCapability -> Aeson.Value -> IO (Either Text Aeson.Value)
executeCapabilityInternal ValidateToolCapability args = do
    case args of
        Aeson.Object obj ->
            case KeyMap.lookup (AesonKey.fromText "path") obj of
                Just (Aeson.String path) -> do
                    result <- BashTools.loadScript silentTracer (Text.unpack path)
                    case result of
                        Left (BashTools.InvalidScriptError _ code _) ->
                            pure $ Right $ Aeson.object
                                [ "valid" .= False
                                , "error" .= ("Script exited with " ++ show code)
                                ]
                        Left (BashTools.InvalidDescriptionError _ jsonErr) ->
                            pure $ Right $ Aeson.object
                                [ "valid" .= False
                                , "error" .= ("JSON parse error: " ++ jsonErr)
                                ]
                        Right desc ->
                            pure $ Right $ Aeson.object
                                [ "valid" .= True
                                , "description" .= desc.scriptInfo
                                ]
                _ -> pure $ Left "Missing or invalid 'path' argument"
        _ -> pure $ Left "Arguments must be a JSON object"

executeCapabilityInternal ScaffoldAgentCapability args = do
    case Aeson.fromJSON args of
        Aeson.Success opts -> do
            case NewAgent.buildAgentConfig opts of
                Left err -> pure $ Left $ Text.pack err
                Right agent -> pure $ Right $ Aeson.object
                    [ "slug" .= agent.slug
                    , "config" .= agent
                    ]
        Aeson.Error err -> pure $ Left $ Text.pack err

executeCapabilityInternal ScaffoldToolCapability args = do
    case args of
        Aeson.Object obj -> do
            case (KeyMap.lookup (AesonKey.fromText "slug") obj,
                  KeyMap.lookup (AesonKey.fromText "language") obj) of
                (Just (Aeson.String slug), Just (Aeson.String lang)) ->
                    case Map.lookup (Text.toLower lang) NewTool.supportedLanguages of
                        Nothing -> pure $ Left $ "Unsupported language: " <> lang
                        Just language -> do
                            let template = NewTool.renderToolTemplate language slug
                            pure $ Right $ Aeson.object
                                [ "slug" .= slug
                                , "language" .= lang
                                , "content" .= template.templateContent
                                , "extension" .= template.templateExtension
                                ]
                _ -> pure $ Left "Missing 'slug' or 'language' argument"
        _ -> pure $ Left "Arguments must be a JSON object"

executeCapabilityInternal GetSpecCapability _args = do
    pure $ Right $ Aeson.object
        [ "spec" .= ("bash-tools" :: Text)
        , "documentation" .= bashToolsDocumentation
        ]
  where
    bashToolsDocumentation :: Text
    bashToolsDocumentation = Text.unlines
        [ "# Bash Tools Specification"
        , ""
        , "A bash-tool implements two commands: describe and run"
        , "..."
        ]

silentTracer :: Prod.Tracer IO a
silentTracer = Prod.Tracer (const (pure ()))

formatExecutionTime :: NominalDiffTime -> Double
formatExecutionTime = realToFrac

formatResults :: OperationResult -> ByteString
formatResults result = LByteString.toStrict $ Aeson.encode $ Aeson.object
    [ "capability" .= resultCapability result
    , "data" .= resultData result
    , "executionTime" .= formatExecutionTime (resultExecutionTime result)
    ]
