{-# LANGUAGE DeriveGeneric #-}

module System.Agents.FileSandbox.Predicate where

import Data.Aeson (FromJSON, ToJSON)
import Data.Text (Text)
import GHC.Generics (Generic)
import System.FilePath (isRelative, normalise, takeExtension, takeFileName)
import qualified System.FilePath.Glob as Glob

-- | Path validation error types
data PathError
  = PathNotAllowed FilePath
  | PathNotAbsolute FilePath
  | PathOutsideSandbox FilePath
  | PathIOError FilePath String
  | FileTooLarge FilePath Integer Integer  -- path, actual, max
  | InvalidExtension FilePath String [String]  -- path, actual, allowed
  deriving (Show, Eq, Generic)

instance FromJSON PathError
instance ToJSON PathError

-- | Path predicate for expressing file access rules.
-- 
-- Predicates are evaluated against the *symlink path* (not resolved).
-- TODO: Add FollowSymlinks / SymlinkTarget variants for resolved-path checking.
data PathPredicate
  = -- | Exact file path match (after canonicalization)
    FileExactly FilePath
    
    -- | Exact directory match (contents not included unless specified)
  | DirectoryExactly FilePath
    
    -- | Directory and all subdirectories recursively
  | DirectoryRecursive FilePath
    
    -- | Directory contents only, NOT subdirectories
  | DirectoryShallow FilePath
    
    -- | Glob pattern match on filename (uses System.FilePath.Glob syntax)
  | FilePattern String
    
    -- | File extension in allowed list (e.g., ["hs", "json"])
  | FileExtension [String]
    
    -- | Maximum file size in bytes (for read operations)
  | FileSizeLessThan Integer
    
    -- | Path is within a subdirectory of the given path
  | ChildOf FilePath
    
    -- | Logical combinators
  | And PathPredicate PathPredicate
  | Or PathPredicate PathPredicate
  | Not PathPredicate
  | Any [PathPredicate]  -- Convenience: Or of all
  | All [PathPredicate]  -- Convenience: And of all
    
    -- | Always allow (use sparingly)
  | AlwaysAllow
    
    -- | Always deny (secure default, explicit)
  | AlwaysDeny
  deriving (Show, Eq, Generic)

instance FromJSON PathPredicate
instance ToJSON PathPredicate

-- 
-- The path should be absolute and canonicalized before evaluation.

### 2. FileSandbox Resource Module

Create `System.Agents.FileSandbox`:

```haskell
module System.Agents.FileSandbox where

import Control.Concurrent.STM
import Control.Exception (try)
import qualified Data.HashMap.Strict as HashMap
import Data.Time (UTCTime, getCurrentTime)
import System.Directory (canonicalizePath, doesFileExist, doesDirectoryExist, getFileSize)

import System.Agents.FileSandbox.Predicate
import System.Agents.OS.Core.Types (ResourceId(..), newEntityId)
import System.Agents.OS.Resources.Types

-- | Configuration for a file sandbox
data FileSandboxConfig = FileSandboxConfig
  { sandboxPredicate :: PathPredicate
  , sandboxMaxFileSize :: Maybe Integer  -- bytes, Nothing = no limit
  , sandboxName :: Maybe Text
  } deriving (Show, Eq, Generic)

instance FromJSON FileSandboxConfig
instance ToJSON FileSandboxConfig

-- | Runtime file sandbox resource
data FileSandbox = FileSandbox
  { sandboxId :: ResourceId
  , sandboxConfig :: FileSandboxConfig
  , sandboxCreatedAt :: UTCTime
  } deriving (Show)

-- | Result of file access validation
data AccessResult
  = AccessGranted
  | AccessDenied PathError
  deriving (Show, Eq)

-- | Validate a file read operation against a sandbox.
-- This checks both the predicate and file size limits.
validateFileRead :: FileSandbox -> FilePath -> IO AccessResult
validateFileRead sandbox path = do
  -- 1. Canonicalize path (but keep symlink info)
  canonical <- canonicalizePath path
  
  -- 2. Check predicate (applied to canonical symlink path)
  predResult <- evaluatePredicate (sandboxConfig sandbox).sandboxPredicate canonical
  case predResult of
    Left err -> pure $ AccessDenied err
    Right () -> do
      -- 3. Check file size if limit configured
      case sandboxConfig sandbox.sandboxMaxFileSize of
        Nothing -> pure AccessGranted
        Just maxSize -> do
          size <- getFileSize canonical
          if size > maxSize
            then pure $ AccessDenied $ FileTooLarge canonical size maxSize
            else pure AccessGranted

-- | Validate a file write operation against a sandbox.
-- For writes, we check parent directory permissions.
validateFileWrite :: FileSandbox -> FilePath -> IO AccessResult
validateFileWrite sandbox path = do
  canonical <- canonicalizePath path
  -- Check if file exists - if so, need write permission on file
  -- If not, need write permission on parent directory
  exists <- doesFileExist canonical
  if exists
    then checkFilePermission canonical
    else checkParentPermission canonical
  where
    checkFilePermission = evaluatePredicate (sandboxConfig sandbox).sandboxPredicate
    checkParentPermission p = do
      let parent = takeDirectory p
      evaluatePredicate (sandboxConfig sandbox).sandboxPredicate parent

-- | Create a file sandbox resource in the OS registry.
createFileSandbox :: ResourceContext -> FileSandboxConfig -> IO ResourceId
createFileSandbox ctx config = do
  eid <- newEntityId
  let rid = ResourceId eid
  createdAt <- getCurrentTime
  let sandbox = FileSandbox
        { sandboxId = rid
        , sandboxConfig = config
        , sandboxCreatedAt = createdAt
        }
  -- Register as OS resource
  let handle = ResourceHandle
        { handleId = rid
        , handleCleanup = pure ()  -- Sandboxes are stateless
        , handleAccess = \f -> f (FileSandboxAccessor sandbox)
        }
  let info = ResourceInfo
        { resourceId = rid
        , resourceScope = ResourceScope (contextScope ctx)
        , resourceType = FileSandboxResourceType config
        , resourceCreatedAt = createdAt
        }
  atomically $ registerResourceHandleSTM (contextRegistry ctx) handle info
  pure rid

-- | Look up a file sandbox by ID.
lookupFileSandbox :: ResourceRegistry -> ResourceId -> IO (Maybe FileSandbox)
lookupFileSandbox registry rid = do
  mHandle <- lookupResourceHandle registry rid
  case mHandle of
    Nothing -> pure Nothing
    Just handle -> do
      mAccessor <- handleAccess handle $ \acc -> pure $ unFileSandboxAccessor acc
      pure $ fmap sandboxFromAccessor mAccessor
```

## Security Considerations

1. **Default Deny**: All file access is denied unless explicitly allowed
2. **Symlink Handling**: Predicates apply to symlink path (not resolved). TODO for resolved-path variant.
3. **Path Canonicalization**: All paths are canonicalized before predicate evaluation
4. **Immutable Sandboxes**: Sandboxes are immutable after creation (no dynamic updates)
5. **Size Limits**: Optional file size limits prevent resource exhaustion
6. **Extension Filtering**: Can restrict to specific file types
