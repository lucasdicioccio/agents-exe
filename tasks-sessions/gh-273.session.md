{- | Initialize a Lua toolbox from a description.

This function:
1. Creates a new Lua state using hslua
2. Configures the sandbox (removes dangerous functions)
3. Sets up memory limit tracking
4. Creates an MVar lock for serializing access
5. Returns a 'Toolbox' ready for script execution

Returns an error if the Lua state cannot be created.
-}
initializeToolbox ::
    Tracer IO Trace ->
    LuaToolboxDescription ->
    IO (Either String Toolbox)
initializeToolbox tracer desc = do
    runTracer tracer (StateInitializedTrace desc.luaToolboxName)

    -- Create Lua state
    result <- try $ Lua.newstate

    case result of
        Left (e :: SomeException) -> do
            let errMsg = "Failed to create Lua state: " <> show e
            pure $ Left errMsg
        Right lstate -> do
            -- Configure sandbox
            configureSandbox lstate

            -- Set up memory limit if configured
            when (desc.luaToolboxMaxMemoryMB > 0) $
                applyMemoryLimit lstate desc.luaToolboxMaxMemoryMB

            -- Create lock for serializing access
            lock <- newEmptyMVar
            putMVar lock ()

            pure $ Right Toolbox
                { toolboxLuaState = lstate
                , toolboxConfig = desc
                , toolboxLock = lock
                , toolboxName = desc.luaToolboxName
                }

{- | Close a toolbox and release its resources.

This function closes the Lua state and should be called when the
sandbox is no longer needed.
-}
closeToolbox :: Tracer IO Trace -> Toolbox -> IO ()
closeToolbox tracer toolbox = do
    Lua.close (toolboxLuaState toolbox)
    runTracer tracer StateClosedTrace

{- | Configure the Lua sandbox by removing dangerous functions.

This removes access to:
* os.execute - could execute arbitrary shell commands
* os.remove - could delete files
* os.rename - could move files
* io.popen - could execute shell commands
* loadfile/dofile - could load arbitrary Lua files
* require - could load arbitrary modules
* package.loadlib - could load native libraries

The safe subset includes:
* Basic types and operators
* String manipulation (string library)
* Table manipulation (table library)
* Math functions (math library)
* Coroutine support (coroutine library)
* Error handling (pcall, xpcall)
* Basic io (print, tostring, tonumber, etc.)
-}
configureSandbox :: LuaState -> IO ()
configureSandbox lstate = Lua.runWith lstate $ do
    Lua.openlibs  -- First load all standard libraries

    -- Then remove dangerous functions from global namespace
    removeGlobals
        [ "dofile"
        , "loadfile"
        , "require"
        ]

    -- Remove dangerous functions from 'os' table
    withTable "os" $ removeFields
        [ "execute"
        , "remove"
        , "rename"
        , "exit"
        , "setlocale"
        , "getenv"  -- could leak sensitive environment variables
        ]

    -- Remove dangerous functions from 'io' table
    withTable "io" $ removeFields
        [ "popen"
        , "tmpfile"
        ]

    -- Remove package table entirely (prevents module loading)
    Lua.pushglobaltable
    Lua.pushstring "package"
    Lua.pushnil
    Lua.settable (Lua.nthFromTop 3)
    Lua.pop 1
  where
    removeGlobals :: [Text] -> Lua ()
    removeGlobals names = mapM_ removeGlobal names

    removeGlobal :: Text -> Lua ()
    removeGlobal name = do
        Lua.pushglobaltable
        Lua.pushstring name
        Lua.pushnil
        Lua.settable (Lua.nthFromTop 3)
        Lua.pop 1

    withTable :: Text -> Lua () -> Lua ()
    withTable name action = do
        Lua.getglobal name
        action
        Lua.pop 1

    removeFields :: [Text] -> Lua ()
    removeFields names = mapM_ removeField names

    removeField :: Text -> Lua ()
    removeField name = do
        Lua.pushstring name
        Lua.pushnil
        Lua.settable (Lua.nthFromTop 3)

{- | Execute a Lua script in the sandbox.

3. Loads and executes the script

5. Returns the result or an error

{- | Execute a Lua script with access to the tool portal.

This is the full execution function that:
1. Sets up the tool portal as a global Lua function
2. Executes the script with the portal available
3. Handles timeouts and memory limits
4. Provides detailed error information

The portal is exposed to Lua as a 'call_tool' function that takes
a tool name and arguments table.
-}
executeScriptWithPortal ::
    Toolbox ->
    Text ->  -- ^ Lua script source
    Maybe ToolPortal ->  -- ^ Optional tool portal for calling other tools
    [Text] ->  -- ^ Allowed tools whitelist
    IO (Either ScriptError ExecutionResult)
executeScriptWithPortal toolbox script mPortal allowedTools = do
    -- Acquire lock to serialize access within this toolbox instance
    withMVar (toolboxLock toolbox) $ \() -> do
        executeScriptInternal toolbox script mPortal allowedTools

executeScriptInternal ::
    Toolbox ->
    Text ->
    Maybe ToolPortal ->
    [Text] ->
    IO (Either ScriptError ExecutionResult)
executeScriptInternal toolbox script mPortal allowedTools = do
    startTime <- getCurrentTime
    let lstate = toolboxLuaState toolbox
    let maxTime = (toolboxConfig toolbox).luaToolboxMaxExecutionTimeSeconds

    -- Set up portal if provided
    case mPortal of
        Just portal -> setupPortal lstate portal allowedTools
        Nothing -> pure ()

    -- Execute with timeout
    result <- applyTimeout maxTime $ do
        Lua.runWith lstate $ do
            -- Load the script
            status <- Lua.loadbuffer (Text.encodeUtf8 script) "=user"
            if status /= Lua.OK
                then do
                    errMsg <- Lua.tostring' Lua.stackTop
                    Lua.pop 1
                    pure $ Left $ LuaRuntimeError (Text.decodeUtf8 errMsg)
                else do
                    -- Execute the loaded chunk
                    callResult <- pcallWithTraceback 0 Lua.multret
                    if fst callResult /= Lua.OK
                        then do
                            let errMsg = snd callResult
                            pure $ Left $ LuaRuntimeError errMsg
                        else do
                            -- Convert result to JSON
                            nrets <- Lua.gettop
                            jsonValue <- if nrets > 0
                                then luaToJsonValue
                                else pure Aeson.Null
                            Lua.pop nrets
                            pure $ Right jsonValue

    endTime <- getCurrentTime
    let execTime = diffUTCTime endTime startTime

    case result of
        Nothing ->
            -- Timeout occurred
            pure $ Left $ TimeoutError maxTime
        Just (Left err) ->
            pure $ Left err
        Just (Right val) ->
            pure $ Right ExecutionResult
                { resultValue = val
                , resultExecutionTime = execTime
                }

-- | Helper to call Lua function with stack trace capture
pcallWithTraceback :: Int -> Int -> Lua (Lua.Status, Text)
pcallWithTraceback nargs nresults = do
    -- Get debug.traceback for error handling
    Lua.getglobal "debug"
    Lua.getfield Lua.stackTop "traceback"
    Lua.remove (Lua.nthFromTop 2)  -- remove 'debug' table

    -- Insert traceback function before function to be called
    Lua.insert (Lua.nthFromTop (2 + nargs))

    -- Call with error handler
    status <- Lua.pcall nargs nresults (Lua.nthFromTop (-(2 + nargs)))

    if status /= Lua.OK
        then do
            errMsg <- Lua.tostring' Lua.stackTop
            Lua.pop 1
            pure (status, Text.decodeUtf8 errMsg)
        else
            pure (status, "")

{- | Convert the value at the top of the Lua stack to JSON.

Handles:
* nil -> Null
* boolean -> Bool
* number -> Number (integer if whole number)
* string -> String
* table -> Object or Array (detected by keys)
-}
luaToJsonValue :: Lua Aeson.Value
luaToJsonValue = do
    ltype <- Lua.ltype Lua.stackTop
    case ltype of
        Lua.TypeNil -> do
            Lua.pop 1
            pure Aeson.Null
        Lua.TypeBoolean -> do
            b <- Lua.toboolean Lua.stackTop
            Lua.pop 1
            pure $ Aeson.Bool b
        Lua.TypeNumber -> do
            n <- Lua.tonumber Lua.stackTop
            Lua.pop 1
            -- Check if it's an integer
            let intVal = round n :: Int
            if fromIntegral intVal == n
                then pure $ Aeson.Number (fromIntegral intVal)
                else pure $ Aeson.Number (fromRational (toRational n))
        Lua.TypeString -> do
            s <- Lua.tostring' Lua.stackTop
            Lua.pop 1
            pure $ Aeson.String (Text.decodeUtf8 s)
        Lua.TypeTable -> do
            convertTable
        _ -> do
            -- For other types (function, userdata, thread, lightuserdata),
            -- convert to string representation
            Lua.pushstring "__tostring"
            Lua.gettable (Lua.nthFromTop 2)
            if Lua.istable (Lua.nthFromTop 2)
                then do
                    -- Call tostring metamethod if exists
                    Lua.insert (Lua.nthFromTop 2)
                    _ <- Lua.pcall 1 1 Nothing
                    s <- Lua.tostring' Lua.stackTop
                    Lua.pop 1
                    pure $ Aeson.String (Text.decodeUtf8 s)
                else do
                    Lua.pop 1  -- pop __tostring
                    s <- Lua.tostring' Lua.stackTop
                    Lua.pop 1
                    pure $ Aeson.String (Text.decodeUtf8 s)

-- | Convert a Lua table to JSON (object or array)
convertTable :: Lua Aeson.Value
convertTable = do
    -- Check if it's an array (integer keys starting at 1, contiguous)
    isArr <- isArray
    if isArr
        then convertArray
        else convertObject

-- | Check if the table at top of stack is an array
isArray :: Lua Bool
isArray = do
    Lua.pushnil  -- first key
    isSequential 1
  where
    isSequential :: Int -> Lua Bool
    isSequential expectedIdx = do
        hasNext <- Lua.next (Lua.nthFromTop 2)
        if not hasNext
            then pure True
            else do
                -- Check if key is the expected integer
                isNum <- Lua.isnumber Lua.stackTop
                if isNum
                    then do
                        idx <- Lua.tointeger Lua.stackTop
                        Lua.pop 2  -- pop key and value
                        if idx == expectedIdx
                            then isSequential (expectedIdx + 1)
                            else pure False
                    else do
                        Lua.pop 2  -- pop key and value
                        pure False

-- | Convert array table to JSON Array
convertArray :: Lua Aeson.Value
convertArray = do
    vals <- collectArrayValues
    pure $ Aeson.Array (Data.Vector.fromList vals)

-- | Collect values from an array table
-- Pop the table when done
collectArrayValues :: Lua [Aeson.Value]
collectArrayValues = do
    let lenIdx = Lua.nthFromTop 1
    Lua.len lenIdx
    len' <- Lua.tointeger Lua.stackTop
    Lua.pop 1

    vals <- mapM (\i -> do
        Lua.pushinteger i
        Lua.gettable (Lua.nthFromTop 2)
        val <- luaToJsonValue  -- This will pop the value
        pure val
        ) [1..len']

    Lua.pop 1  -- pop the table
    pure vals

-- | Convert object table to JSON Object
convertObject :: Lua Aeson.Value
convertObject = do
    -- Pop the table when done
    pairs <- collectObjectPairs
    pure $ Aeson.Object (Data.HashMap.Strict.fromList pairs)

collectObjectPairs :: Lua [(Text, Aeson.Value)]
collectObjectPairs = do
    Lua.pushnil  -- first key
    go []
  where
    go :: [(Text, Aeson.Value)] -> Lua [(Text, Aeson.Value)]
    go acc = do
        hasNext <- Lua.next (Lua.nthFromTop 2)
        if not hasNext
            then do
                Lua.pop 1  -- pop the table
                pure acc
            else do
                -- Stack: table, key, value
                key <- Lua.tostring' (Lua.nthFromTop 2)
                val <- luaToJsonValue  -- This pops the value
                -- Stack: table, key
                Lua.pop 1  -- pop key
                -- Stack: table
                go ((Text.decodeUtf8 key, val) : acc)

## Summary

Implemented Phase 2 of the LuaToolbox feature: a core Lua interpreter module using the hslua library.

### New File: `src/System/Agents/Tools/LuaToolbox.hs`

**Core Types:**
- `Trace` - Trace events for monitoring Lua toolbox operations (initialization, execution, errors, etc.)
- `Toolbox` - Runtime state containing Lua state, configuration, and access lock
- `ScriptError` - Error types (LuaRuntimeError, TimeoutError, MemoryError, SandboxError, etc.)
- `ExecutionResult` - Result with JSON value and execution time

**Key Functions:**
- `initializeToolbox` - Creates a sandboxed Lua state with memory limits
- `closeToolbox` - Cleans up Lua state resources
- `executeScript` - Runs Lua code and returns JSON results
- `executeScriptWithPortal` - Full execution with tool portal access
- `configureSandbox` - Removes dangerous functions (os.execute, io.popen, etc.)
- `applyMemoryLimit` - Sets up memory limit tracking
- `applyTimeout` - Enforces execution time limits via Haskell's timeout

**Sandbox Features:**
- Removes dangerous globals: dofile, loadfile, require
- Removes dangerous os.* functions: execute, remove, rename, exit, setlocale, getenv
- Removes dangerous io.* functions: popen, tmpfile
- Removes entire 'package' table to prevent module loading
- Preserves safe functionality: math, string, table, coroutine, pcall/xpcall

**Lua-to-JSON Conversion:**
- nil → null
- boolean → boolean
- number → number (integer detection)
- string → string
- table → array (sequential integer keys) or object (string keys)

**Dependencies Added to `agents.cabal`:**
- `hslua >= 2.3`
- `hslua-marshalling`
