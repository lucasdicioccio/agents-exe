-- | Render a conversation tree as a nested list
renderConversationTree :: 
    Set ConversationId  -- ^ Expanded conversation IDs
    -> [ConversationId] -- ^ Currently selected path
    -> [ConversationNode] 
    -> Widget WidgetName
renderConversationTree expanded selectedPath nodes = 
    vBox $ map (renderNode expanded selectedPath 0) nodes

-- | Render a single conversation node at a given depth
renderNode :: 
    Set ConversationId 
    -> [ConversationId] 
    -> Int              -- ^ Indentation depth
    -> ConversationNode 
    -> Widget WidgetName
renderNode expanded selectedPath depth node = 
    let conv = nodeConversation node
        convId = conversationId conv
        isExpanded = Set.member convId expanded
        hasChildren = not (null $ nodeChildren node)
        isSelected = convId `elem` selectedPath
        
        -- Indentation based on depth
        indent = hBox $ replicate depth (str "  ")
        
        -- Expand/collapse indicator
        expander = if hasChildren
            then if isExpanded
                then str "[-] "
                else str "[+] "
            else str "    "
        
        -- Conversation name with visual indicators
        nameWidget = str $ Text.unpack $ conversationName conv
        statusWidget = renderConversationStatus (conversationStatus conv)
        
        -- Apply styles if selected
        baseWidget = hBox [indent, expander, nameWidget, str " ", statusWidget]
        styledWidget = if isSelected
            then withAttr selectedConversationAttr baseWidget
            else baseWidget
        
        -- Render children if expanded
        childrenWidgets = if isExpanded
            then map (renderNode expanded selectedPath (depth + 1)) (nodeChildren node)
            else []
    in 
        vBox (styledWidget : childrenWidgets)

-- | Render conversation status indicator
renderConversationStatus :: ConversationStatus -> Widget WidgetName
renderConversationStatus ConversationStatus_Active = 
    withAttr activeStatusAttr (str "●")
renderConversationStatus ConversationStatus_WaitingForInput = 
    withAttr waitingStatusAttr (str "◐")
renderConversationStatus ConversationStatus_Paused = 
    withAttr pausedStatusAttr (str "◑")
```

handleConversationListEvent :: 
    BrickEvent WidgetName AppEvent 
    -> EventM WidgetName TuiState ()
handleConversationListEvent ev = do
    case ev of
        VtyEvent (V.EvKey (V.KChar '\t') []) -> 
            -- Tab to toggle expansion of selected conversation
            toggleSelectedConversation
        VtyEvent (V.EvKey V.KEnter []) -> 
            -- Enter to view selected conversation
            viewSelectedConversation
        VtyEvent (V.EvKey V.KRight []) -> 
            -- Right arrow to expand
            expandSelectedConversation
        VtyEvent (V.EvKey V.KLeft []) -> 
            -- Left arrow to collapse
            collapseSelectedConversation
        VtyEvent (V.EvKey V.KUp []) -> 
            -- Navigate up in the tree

        VtyEvent (V.EvKey V.KDown []) -> 
            -- Navigate down in the tree
            navigateConversationDown
        _ -> pure ()

-- | Toggle expansion state of the selected conversation
toggleSelectedConversation :: EventM WidgetName TuiState ()
toggleSelectedConversation = do
    mSelectedId <- getSelectedConversationId
    case mSelectedId of
        Nothing -> pure ()
        Just convId -> do
            expanded <- use (tuiUI . conversationTreeExpanded)
            if Set.member convId expanded
                then tuiUI . conversationTreeExpanded .= Set.delete convId expanded
                else tuiUI . conversationTreeExpanded .= Set.insert convId expanded

-- | Navigate to parent conversation
navigateToParent :: EventM WidgetName TuiState ()
navigateToParent = do
    mSelectedId <- getSelectedConversationId
    core <- liftIO . readTVarIO =<< use tuiCore
    
    case mSelectedId of
        Nothing -> pure ()
        Just convId -> do
            -- Find the conversation and check if it has a parent
            case find ((== convId) . conversationId) (coreConversations core) of
                Nothing -> pure ()
                Just conv -> case conversationSession conv >>= parentConversationId of
                    Nothing -> pure ()  -- No parent
                    Just parentId -> do
                        -- Update selected path to parent
                        tuiUI . selectedConversationPath .= [parentId]
```

## Tradeoffs

1. **Screen Real Estate**: Deep trees consume horizontal space. Consider:
   - Maximum depth limit for display
   - Horizontal scrolling for deep trees
   - Collapsing all but the current path

2. **Navigation Complexity**: Tree navigation is more complex than list navigation. The chosen keybindings (arrows + Tab) follow common tree widget conventions.

3. **Performance**: Rendering large trees could be slow. Consider:
   - Virtualization (only render visible nodes)
   - Lazy loading of child conversations
   - Caching the tree structure
