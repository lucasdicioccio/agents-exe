```haskell
testPortalToolCall :: Assertion
testPortalToolCall = do
    -- Create a mock portal that echoes back the call
    let mockPortal call = pure $ ToolResult
            { resultData = Aeson.object
                [ "tool" .= callToolName call
                , "args" .= callArgs call
                ]
            , resultDuration = 0.1
            , resultTraceId = "test-trace-123"
            }

    let desc = testLuaToolbox { luaToolboxAllowedTools = ["test_tool"] }

    result <- withLuaToolbox desc $ \box -> do
        executeScriptWithPortal box
            (Text.unlines
                [ "local tools = require('tools')"
                , "local result = tools.call('test_tool', {x = 1})"
                , "return result"
                ])
            (Just mockPortal)
            ["test_tool"]

    case result of
        Left err -> assertFailure $ show err
        Right execResult -> do
            let Aeson.Object hm = execResult.resultValue
            case KeyMap.lookup "data" hm >>= parseData of
                Just (tool, args) -> do
                    tool @?= "test_tool"
                    args @?= Aeson.object ["x" .= (1 :: Int)]
                Nothing -> assertFailure $ "Unexpected result: " ++ show execResult.resultValue
  where
    parseData val = case val of
        Aeson.Object d -> do
            tool <- KeyMap.lookup "tool" d >>= parseString
            args <- KeyMap.lookup "args" d
            pure (tool, args)
        _ -> Nothing
    parseString (Aeson.String s) = Just s
    parseString _ = Nothing
```

## Available Modules

### json

Encode/decode JSON:

```lua
local json = require("json")
local str = json.encode({name = "test", value = 42})
local obj = json.decode('{"x": 1}')
```

### text

UTF-8 string operations:

```lua
local text = require("text")
text.split("a,b,c", ",")  -- {"a", "b", "c"}
text.trim("  hello  ")     -- "hello"
text.startswith("hello", "he")  -- true
```

### time

Time functions:

```lua
local time = require("time")
local now = time.now()
time.sleep(2)
local str = time.format(now, "%Y-%m-%d")
```

### fs

Sandboxed filesystem access:

```lua
local fs = require("fs")
local content, err = fs.read("./data/file.txt")
fs.write("./data/output.txt", "hello")
local exists = fs.exists("./data")
```

### http

HTTP requests (host-restricted):

```lua
local http = require("http")
local result = http.get("http://localhost:3000/api")
-- result.status, result.headers, result.body
```

### tools

Call other tools through the portal:

```lua
local tools = require("tools")

-- Generic call
local result = tools.call("bash", {command = "ls -la"})

-- Convenience wrappers
local r = tools.bash.run("echo hello")
print(r.data.stdout)
```

## Security

The Lua sandbox:
- Blocks `os.execute`, `io.popen`, `loadfile`, `dofile`, `require`
- Enforces memory limits
- Enforces execution timeouts
- Restricts filesystem access to `allowedPaths`
- Restricts HTTP to `allowedHosts`
- Only allows calling whitelisted tools

## Error Handling

Use `pcall` for error handling:

```lua
local ok, result = pcall(function()
    return tools.bash.run("rm -rf /")
end)

if not ok then
    print("Error: " .. result)
end
```
